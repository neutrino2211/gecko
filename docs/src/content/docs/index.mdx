## Quick Example

```ts
package main

declare external variardic func printf(format: string): int

class Point {
    let x: int
    let y: int

    func new(x: int, y: int): Point {
        let p: Point
        p.x = x
        p.y = y
        return p
    }

    func add(self, other: Point): Point {
        return Point::new(self.x + other.x, self.y + other.y)
    }
}

func main(): int {
    let a: Point = Point::new(1, 2)
    let b: Point = Point::new(3, 4)
    let c: Point = a.add(b)
    printf("Result: (%d, %d)\n", c.x, c.y)
    return 0
}
```
